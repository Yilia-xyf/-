获取当前页面隔天过来的中文参数的方法
    var localParam = window.location.search;
    function GetQueryString(str){
        var rs=new RegExp("(^|)"+str+"=([^&]*)(&|$)","gi").exec(localParam),tmp;
        if(tmp=rs){
            return tmp[2];
        }else{
            return "没有这个参数";
        }
    }
